private int compute_size_of_tree(Node<Integer> root){
    Stack<Node<Integer>> stack = new Stack<>();
    stack.push(root);
    int res = 0;
    while (stack.size() > 0) {
        Node<Integer> subtree = stack.pop();
        if (subtree != null) {
            res ++;
            stack.push(subtree.prev);
            stack.push(subtree.next);
        }
    }
    return res;   
}


public ArrayList<Integer> output_sorted_array_of_two_trees(Node<Integer> root1 , Node<Integer> root2){

    assert root1 != null && root2 != null : "tree can't be empty";
    
    ArrayList<Integer> result_array = new ArrayList<>();
    
    Node<Integer> smaller = new Node<>();
    Node<Integer> bigger = new Node<>();
    
    if( compute_size_of_tree(root1) >= compute_size_of_tree(root2) ){
        smaller = root2;
        bigger = root1;
    }
    else{
        smaller = root1;
        bigger = root2;
    }
    
    Stack<Node<Integer>> stack = new Stack<>();
    stack.push(smaller);
    int res = 0;
    while(stack.size() > 0){
        Node<Integer> subtree = stack.pop();
        if(subtree != null){
            if( subtree.value.search_in_tree(bigger) ){ 
                result_array.add(subtree.value);
            }
        }
    }
    
    return Collection.sort(result_array);
}

private Node<Integer> smallest_Node_BiggerEqual_than_number(Node<Integer> root , int l){
    
    assert root != null;
    
    if( (root.value <= l) && (root.next.value == null || root.next.value > l)  ){
        return root;
    }
    else{
        if(root.value < l){
            if(max_of_tree(root).value < l){    //this method is almost the same as Exercise 3 , in AVL tree , the worst case is almost O(logn)
                return max_of_tree(root);  
            }
            else{
                return smallest_Node_BiggerEqual_than_number(root.next , l);
            }
        }
        
        else{
            if(min_of_tree(root).value >= l){
                return min_of_tree(root);       //this method is almost the same as Exercise 3 , in AVL tree , the worst case is almost O(logn)
            }
            else{
                return smallest_Node_BiggerEqual_than_number(root.prev , l);
            }
        }
    }
}


private Node<Integer> find_smallest_bigger_one(Node<Integer> node){
    if(node.next != null){
        Node y = node.next;
        while(y.prev != null){
            y = y.prev;
        }
        return y;
    }
    
    else{
        Node y = node.parent
        while(y != null && node == y.right){
            node = y;
            y = y.parent;
        }
        return y;
    }
}



public int sum_of_the_value_in_interval(int l , int r , Node<Integer> A){
    
    Node<Integer> node_smaller = smallest_Node_BiggerEqual_than_number(A , l);
    
    int sum = node_smaller.value;
    
    while(node_smaller != null && node_smaller.value <= r){
        sum += node_smaller.value;
        node_smaller = find_smallest_bigger_one(node_smaller);
    }
    
    return sum;
}


COMMENT:


in our code , we have 3 methods . method 1 : "smallest_Node_BiggerEqual_than_number" : we are using recursive to do , 
                                             and the method "max_of_tree(root).value" and "min_of_tree(root)" take logn in AVL trees , so the time complexity is O(logn)
                                  
                                  method 2 : "find_smallest_bigger_one" this is O(1)
                                  
                                  method 3 : "sum_of_the_value_in_interval" , by hypothesis , all node in this tree without duplicates , so in worst case it take O(r-l)
                                  
                                  
                                  Therefore , our Algorithm has the stated worse case time complexity O(logn + (r-l)).

public class find_PreImage{
    
    private ArrayList<Integer> sorted_array = new ArrayList<>();
    
    private int n;
    
    public find_PreImage (function f , int n){
        
        assert n >= 1 : "n must in the nature number";
        
        this.n = n;
        
        for( int i = 1 ; i < n+1 ; i++ ){              //at the end of the for loop , we will get a sorted array , because function f is strictly increasing. and this take almost O(n).
            sorted_array.add(f(i));  
        }   
    }





    public int output_pre_image(int j){
        
         //we will do binary search
         
         int low = 0 , high = n-1;
         
         while(low <= high){
             int mid = low + (high - low)/2;
             if( sorted_array.get(mid).eauals(j) ){
                 return mid + 1;
             }
             else{
                 if(sorted_array.get(mid) < j){
                     low = mid + 1;
                 }
                 else{
                     high = mid - 1;
                 }
             }
         }    
         
         return -1;         //actually we will not reach here, because array is sorted and use binary search will find the value
    }


}
