class Node<Integer>{
    Integer element;
    Node<Integer> prev;
    Node<Integer> next;
}

===================================================================

//our Algorithm is as follows:

===================================================================

public int Sum_of_min_and_max_of_the_tree(Node<Integer> root){

    assert root != null : "the tree must exist";
    
    Node<Integer> minimal = root;
    
    Node<Integer> maximal = root;
    
    while(minimal.prev != null){
        minimal = minimal.prev;    
    }
    
    while(maximal.next != null){
        maximal = maximal.next;
    }
    
    return (minimal.element + maximal.element);
}
